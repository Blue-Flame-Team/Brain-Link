+  static const String _favoritesKey = 'favorites_ids';
+
+
+  // ===================== دوال المفضلات الجديدة =====================
+
+  static Future<void> saveFavoritesIds(List<String> ids) async {
+    final prefs = await SharedPreferences.getInstance();
+    await prefs.setStringList(_favoritesKey, ids);
+  }
+
+  static Future<List<String>> getFavoritesIds() async {
+    final prefs = await SharedPreferences.getInstance();
+    return prefs.getStringList(_favoritesKey) ?? [];
+  }
+
+  static Future<void> addFavoriteId(String id) async {
+    List<String> ids = await getFavoritesIds();
+    if (!ids.contains(id)) {
+      ids.add(id);
+      await saveFavoritesIds(ids);
+    }
+  }
+
+  static Future<void> removeFavoriteId(String id) async {
+    List<String> ids = await getFavoritesIds();
+    ids.remove(id);
+    await saveFavoritesIds(ids);
+  }
+
+  static Future<bool> isFavorite(String id) async {
+    List<String> ids = await getFavoritesIds();
+    return ids.contains(id);
+  }
